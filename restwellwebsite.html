    function setActive(id){

      // highlight current

      document.querySelectorAll('.navlinks a').forEach(a=>a.classList.remove('active'));

      const link = Array.from(document.querySelectorAll('.navlinks a')).find(a=>a.getAttribute('href')===`#/${id}`);

      if(link) link.classList.add('active');

    }


    function route(id){

      if(!id) id = (location.hash.replace('#/','') || 'home');

      const view = VIEWS[id] || VIEWS['home'];

      document.getElementById('app').innerHTML = view();

      setActive(id);

      history.replaceState(null,'',`#/${id}`);

      window.scrollTo({top:0,behavior:'smooth'});
